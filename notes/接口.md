```go
func New(typ Type) Value {
   if typ == nil {
      panic("reflect: New(nil)")
   }
   t := typ.(*rtype)
   ptr := unsafe_New(t)
   fl := flag(Ptr)
   return Value{t.ptrTo(), ptr, fl}
}

// implemented in package runtime
```
